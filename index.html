public class Metodos {
public static void main(String args[]){
int z;
sumar();
desplegar(10);
z = cuadrado(4);
System.out.println(z);
raizCuadrada(100);
}

public static void sumar() {
int n1,n2,suma;
Scanner teclado= new Scanner(System.in);
System.out.print("Valor 1 : ");
n1 = teclado.nextInt();
System.out.print("Valor 2 : ");
n2 = teclado.nextInt();
suma = n1 + n2;
System.out.println("La suma es : " + suma);
}

public static void desplegar (int x) {
int i;
for (i=1;i<=x;i++) {
System.out.println("Sonora");
}
}
public static void raizCuadrada(double f){
System.out.print("Raíz cuadrada: ");
System.out.println(Math.sqrt(f));
}

public static int cuadrado (int x) {
int resultado;
resultado = x * x;
return resultado;
}
